# Task Service

Сервис для управления задачами с HTTP API на Go.

## Требования

- Go `1.23+`
- Docker и Docker Compose

## Быстрый запуск через Docker Compose

```bash
docker compose up --build
```

После запуска сервис будет доступен по адресу `http://localhost:8080`.

Если `postgres` уже запускался ранее со старой схемой, пересоздай volume:

```bash
docker compose down -v
docker compose up --build
```

Причина в том, что SQL-файл из `migrations/0001_create_tasks.up.sql` монтируется в `docker-entrypoint-initdb.d` и применяется только при инициализации пустого data volume.

## Swagger

Swagger UI:

```text
http://localhost:8080/swagger/
```

OpenAPI JSON:

```text
http://localhost:8080/swagger/openapi.json
```

## API

Базовый префикс API:

```text
/api/v1
```

Основные маршруты:

- `POST /api/v1/tasks`
- `GET /api/v1/tasks`
- `GET /api/v1/tasks/{id}`
- `PUT /api/v1/tasks/{id}`
- `DELETE /api/v1/tasks/{id}`

func main() {
	logger := slog.New(slog.NewTextHandler(os.Stdout, &slog.HandlerOptions{
		Level: slog.LevelInfo,
	}))

	cfg := loadConfig()

	ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	pool, err := infrastructurepostgres.Open(ctx, cfg.DatabaseDSN)
	if err != nil {
		logger.Error("open postgres", "error", err)
		os.Exit(1)
	}
	defer pool.Close()

	migrator := migrations.NewMigrator(pool, cfg.MigrationsPath)
	if err := migrator.Up(ctx); err != nil {
		logger.Error("run migrations", "error", err)
		os.Exit(1)
	}

	taskRepo := postgresrepo.New(pool)
	recurrenceRepo := postgresrepo.NewRecurrenceRepository(pool)

	taskUsecase := task.NewService(taskRepo)

	validator := recurrencedomain.NewValidator()
	calculator := recurrencedomain.NewCalculator()
	recurrenceUsecase := recurrence.NewService(recurrenceRepo, taskRepo, validator, calculator)

	scheduler := recurrence.NewScheduler(recurrenceRepo, taskRepo, calculator, 1*time.Minute)

	taskHandler := httphandlers.NewTaskHandler(taskUsecase)
	recurrenceHandler := httphandlers.NewRecurrenceHandler(recurrenceUsecase)
	docsHandler := swaggerdocs.NewHandler()
	router := transporthttp.NewRouter(taskHandler, recurrenceHandler, docsHandler)

	server := &http.Server{
		Addr:              cfg.HTTPAddr,
		Handler:           router,
		ReadHeaderTimeout: 5 * time.Second,
	}

	go scheduler.Start(ctx)

	go func() {
		<-ctx.Done()

		logger.Info("shutting down...")
		scheduler.Stop()

		shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
		defer cancel()

		if err := server.Shutdown(shutdownCtx); err != nil {
			logger.Error("shutdown http server", "error", err)
		}
	}()

	logger.Info("http server started", "addr", cfg.HTTPAddr)

	if err := server.ListenAndServe(); err != nil && err != http.ErrServerClosed {
		logger.Error("listen and serve", "error", err)
		os.Exit(1)
	}
}

package recurrence

import (
	"encoding/json"
	"fmt"
	"sort"
	"time"
)

type RecurrenceCalculator interface {
	NextOccurrence(config *RecurrenceConfig, from time.Time) (time.Time, error)
}

type Calculator struct{}

func NewCalculator() *Calculator {
	return &Calculator{}
}

func (c *Calculator) NextOccurrence(config *RecurrenceConfig, from time.Time) (time.Time, error) {
	if config == nil {
		return time.Time{}, fmt.Errorf("recurrence config cannot be nil")
	}

	if from.Before(config.StartDate) {
		from = config.StartDate
	}

	if !config.EndDate.IsZero() && !from.Before(config.EndDate) {
		return time.Time{}, fmt.Errorf("recurrence has ended")
	}

	var next time.Time
	var err error

	switch config.RecurrenceType {
	case RecurrenceDaily:
		next, err = c.calculateDaily(config, from)
	case RecurrenceMonthly:
		next, err = c.calculateMonthlyDate(config, from)
	case RecurrenceEven:
		next, err = c.calculateMonthlyEven(config, from)
	case RecurrenceOdd:
		next, err = c.calculateMonthlyOdd(config, from)
	case RecurrenceDate:
		next, err = c.calculateSpecificDates(config, from)
	default:
		return time.Time{}, fmt.Errorf("unknown recurrence type: %s", config.RecurrenceType)
	}

	if err != nil {
		return time.Time{}, err
	}

	if !config.EndDate.IsZero() && next.After(config.EndDate) {
		return time.Time{}, fmt.Errorf("next occurrence exceeds end_date")
	}

	return next, nil
}

func (c *Calculator) calculateDaily(config *RecurrenceConfig, from time.Time) (time.Time, error) {
	if config.IntervalDays < 1 {
		return time.Time{}, fmt.Errorf("interval_days must be >= 1")
	}

	next := from.AddDate(0, 0, config.IntervalDays)

	next = normalizeToStartOfDay(next)

	return next, nil
}

func (c *Calculator) calculateMonthlyDate(config *RecurrenceConfig, from time.Time) (time.Time, error) {
	if config.MonthlyDay < 1 || config.MonthlyDay > 30 {
		return time.Time{}, fmt.Errorf("monthly_day must be between 1 and 30")
	}

	targetDay := config.MonthlyDay
	year := from.Year()
	month := from.Month()

	if from.Day() < targetDay {

		next := trySetDay(year, month, targetDay)
		if !next.IsZero() && next.After(from) {
			return normalizeToStartOfDay(next), nil
		}
	}

	month++
	if month > 12 {
		month = 1
		year++
	}

	next := trySetDay(year, month, targetDay)
	if next.IsZero() {
		return time.Time{}, fmt.Errorf("failed to calculate next monthly date")
	}

	return normalizeToStartOfDay(next), nil
}

func (c *Calculator) calculateMonthlyEven(config *RecurrenceConfig, from time.Time) (time.Time, error) {
	year := from.Year()
	month := from.Month()
	day := from.Day()
	day++

	for {
		if day%2 == 0 {
			lastDay := lastDayOfMonth(year, month)
			if day <= lastDay {
				next := time.Date(year, month, day, 0, 0, 0, 0, from.Location())
				return normalizeToStartOfDay(next), nil
			}
		}

		day++

		lastDay := lastDayOfMonth(year, month)
		if day > lastDay {
			month++
			if month > 12 {
				month = 1
				year++
			}
			day = 2 // Первый четный день
		}

		if year > from.Year()+10 {
			return time.Time{}, fmt.Errorf("failed to calculate next even day")
		}
	}
}

func (c *Calculator) calculateMonthlyOdd(config *RecurrenceConfig, from time.Time) (time.Time, error) {
	year := from.Year()
	month := from.Month()
	day := from.Day()

	day++

	for {
		if day%2 == 1 {
			lastDay := lastDayOfMonth(year, month)
			if day <= lastDay {
				next := time.Date(year, month, day, 0, 0, 0, 0, from.Location())
				return normalizeToStartOfDay(next), nil
			}
		}

		day++

		lastDay := lastDayOfMonth(year, month)
		if day > lastDay {
			month++
			if month > 12 {
				month = 1
				year++
			}
			day = 1
		}

		if year > from.Year()+10 {
			return time.Time{}, fmt.Errorf("failed to calculate next odd day")
		}
	}
}

func (c *Calculator) calculateSpecificDates(config *RecurrenceConfig, from time.Time) (time.Time, error) {
	if config.SpecificDates == "" {
		return time.Time{}, fmt.Errorf("specific_dates is required")
	}

	var dateStrings []string
	if err := json.Unmarshal([]byte(config.SpecificDates), &dateStrings); err != nil {
		return time.Time{}, fmt.Errorf("invalid specific_dates JSON: %w", err)
	}

	if len(dateStrings) == 0 {
		return time.Time{}, fmt.Errorf("specific_dates is empty")
	}

	var dates []time.Time
	for _, dateStr := range dateStrings {
		date, err := time.Parse("2006-01-02", dateStr)
		if err != nil {
			return time.Time{}, fmt.Errorf("invalid date format: %s", dateStr)
		}
		dates = append(dates, normalizeToStartOfDay(date))
	}

	sort.Slice(dates, func(i, j int) bool {
		return dates[i].Before(dates[j])
	})

	fromNormalized := normalizeToStartOfDay(from)
	for _, date := range dates {
		if date.After(fromNormalized) {
			return date, nil
		}
	}

	return time.Time{}, fmt.Errorf("no more occurrences in specific_dates after %s", from.Format("2006-01-02"))
}

func normalizeToStartOfDay(t time.Time) time.Time {
	return time.Date(t.Year(), t.Month(), t.Day(), 0, 0, 0, 0, t.Location())
}

func lastDayOfMonth(year int, month time.Month) int {
	// Первый день следующего месяца минус один день
	firstOfNextMonth := time.Date(year, month+1, 1, 0, 0, 0, 0, time.UTC)
	lastOfMonth := firstOfNextMonth.AddDate(0, 0, -1)
	return lastOfMonth.Day()
}

func trySetDay(year int, month time.Month, targetDay int) time.Time {
	lastDay := lastDayOfMonth(year, month)

	day := targetDay
	if day > lastDay {
		day = lastDay
	}

	return time.Date(year, month, day, 0, 0, 0, 0, time.UTC)
}

type RecurrenceValidator interface {
	Validate(config *RecurrenceConfig) error

type Validator struct {
	v *validator.Validate
}

func NewValidator() *Validator {
	v := validator.New()

	_ = v.RegisterValidation("recurrence_type", validateRecurrenceType)
	_ = v.RegisterValidation("daily_interval_config", validateDailyIntervalConfig)
	_ = v.RegisterValidation("monthly_date_config", validateMonthlyDateConfig)
	_ = v.RegisterValidation("monthly_even_config", validateMonthlyEvenConfig)
	_ = v.RegisterValidation("monthly_odd_config", validateMonthlyOddConfig)
	_ = v.RegisterValidation("specific_dates_config", validateSpecificDatesConfig)
	_ = v.RegisterValidation("date_range", validateDateRange)
	_ = v.RegisterValidation("iso_date_list", validateISODateList)

	return &Validator{v: v}
}

func (val *Validator) Validate(config *RecurrenceConfig) error {
	if config == nil {
		return fmt.Errorf("recurrence config cannot be nil")
	}

	if err := val.v.Struct(config); err != nil {
		return formatValidationErrors(err)
	}

	return validateRecurrenceLogic(config)
}

func validateRecurrenceType(fl validator.FieldLevel) bool {
	recType := fl.Field().String()
	validTypes := map[string]bool{
		string(RecurrenceDaily):   true,
		string(RecurrenceDate):    true,
		string(RecurrenceEven):    true,
		string(RecurrenceOdd):     true,
		string(RecurrenceMonthly): true,
	}
	return validTypes[recType]
}

func validateDailyIntervalConfig(fl validator.FieldLevel) bool {
	parent := fl.Parent().Interface().(RecurrenceConfig)

	if parent.RecurrenceType != RecurrenceDaily {
		return true
	}

	intervalDays := parent.IntervalDays
	if intervalDays == 0 {
		return false
	}

	return intervalDays >= 1 && intervalDays <= 365
}

func validateMonthlyDateConfig(fl validator.FieldLevel) bool {
	parent := fl.Parent().Interface().(RecurrenceConfig)

	if parent.RecurrenceType != RecurrenceMonthly {
		return true
	}

	monthlyDay := parent.MonthlyDay
	if monthlyDay == 0 {
		return false
	}

	return monthlyDay >= 1 && monthlyDay <= 30
}

func validateMonthlyEvenConfig(fl validator.FieldLevel) bool {
	parent := fl.Parent().Interface().(RecurrenceConfig)

	if parent.RecurrenceType != RecurrenceEven {
		return true
	}

	return parent.IntervalDays == 0 &&
		parent.MonthlyDay == 0 &&
		parent.SpecificDates == ""
}

func validateMonthlyOddConfig(fl validator.FieldLevel) bool {
	parent := fl.Parent().Interface().(RecurrenceConfig)

	if parent.RecurrenceType != RecurrenceOdd {
		return true
	}

	return parent.IntervalDays == 0 &&
		parent.MonthlyDay == 0 &&
		parent.SpecificDates == ""

	parent := fl.Parent().Interface().(RecurrenceConfig)

	if parent.RecurrenceType != RecurrenceDate {
		return true
	}

	if parent.SpecificDates == "" {
		return false
	}

	var dates []string

	return err == nil && len(dates) > 0
}

func validateDateRange(fl validator.FieldLevel) bool {
	endDate := fl.Field().Interface().(time.Time)

	if endDate.IsZero() {
		return true
	}

	parent := fl.Parent().Interface().(RecurrenceConfig)
	startDate := parent.StartDate

	return endDate.After(startDate)
}

func validateISODateList(fl validator.FieldLevel) bool {
	dateListStr := fl.Field().String()

	if dateListStr == "" {
		return true
	}

	var dates []string
	if err := json.Unmarshal([]byte(dateListStr), &dates); err != nil {
		return false
	}

	dateRegex := regexp.MustCompile(`^\d{4}-\d{2}-\d{2}$`)

	for _, dateStr := range dates {
		if !dateRegex.MatchString(dateStr) {
			return false
		}
	}

	return true
}

func validateRecurrenceLogic(config *RecurrenceConfig) error {
	if config.NextOccurrenceAt.Before(config.StartDate) {
		return fmt.Errorf("next_occurrence_at must be >= start_date")
	}

	if !config.EndDate.IsZero() && config.NextOccurrenceAt.After(config.EndDate) {
		return fmt.Errorf("next_occurrence_at must be <= end_date")
	}

	switch config.RecurrenceType {
	case RecurrenceDaily:
		return validateDailyIntervalLogic(config)
	case RecurrenceMonthly:
		return validateMonthlyDateLogic(config)
	case RecurrenceDate:
		return validateSpecificDatesLogic(config)
	}

	return nil
}

func validateDailyIntervalLogic(config *RecurrenceConfig) error {
	if config.IntervalDays == 0 {
		return fmt.Errorf("interval_days is required for DAILY_INTERVAL type")
	}

	if config.IntervalDays < 1 {
		return fmt.Errorf("interval_days must be >= 1, got %d", config.IntervalDays)
	}

	return nil
}

func validateMonthlyDateLogic(config *RecurrenceConfig) error {
	if config.MonthlyDay == 0 {
		return fmt.Errorf("monthly_day is required for MONTHLY_DATE type")
	}

	if config.MonthlyDay < 1 || config.MonthlyDay > 30 {
		return fmt.Errorf("monthly_day must be between 1 and 30, got %d", config.MonthlyDay)
	}

	return nil
}

func validateSpecificDatesLogic(config *RecurrenceConfig) error {
	if config.SpecificDates == "" {
		return fmt.Errorf("specific_dates is required for SPECIFIC_DATES type")
	}

	var dates []string
	if err := json.Unmarshal([]byte(config.SpecificDates), &dates); err != nil {
		return fmt.Errorf("specific_dates must be valid JSON array: %w", err)
	}

	if len(dates) == 0 {
		return fmt.Errorf("specific_dates must contain at least one date")
	}

	for _, dateStr := range dates {
		date, err := time.Parse("2006-01-02", dateStr)
		if err != nil {
			return fmt.Errorf("invalid date format: %s (expected YYYY-MM-DD)", dateStr)
		}

		if date.Before(config.StartDate) {
			return fmt.Errorf("specific date %s must be >= start_date", dateStr)
		}

		if !config.EndDate.IsZero() && date.After(config.EndDate) {
			return fmt.Errorf("specific date %s must be <= end_date", dateStr)
		}
	}

	return nil
}

func formatValidationErrors(err error) error {
	if err == nil {
		return nil
	}

	validationErrors := err.(validator.ValidationErrors)
	var messages []string

	for _, fe := range validationErrors {
		msg := formatFieldError(fe)
		messages = append(messages, msg)
	}

	return fmt.Errorf("validation failed: %s", strings.Join(messages, "; "))
}

func formatFieldError(fe validator.FieldError) string {
	field := fe.Field()
	tag := fe.Tag()
	param := fe.Param()

	switch tag {
	case "required":
		return fmt.Sprintf("%s is required", field)
	case "numeric":
		return fmt.Sprintf("%s must be numeric", field)
	case "gte":
		return fmt.Sprintf("%s must be >= %s", field, param)
	case "lte":
		return fmt.Sprintf("%s must be <= %s", field, param)
	case "ltefield":
		return fmt.Sprintf("%s must be < %s", field, param)
	case "recurrence_type":
		return fmt.Sprintf("%s is not a valid recurrence type", field)
	case "daily_interval_config":
		return fmt.Sprintf("%s configuration is invalid for DAILY_INTERVAL", field)
	case "monthly_date_config":
		return fmt.Sprintf("%s must be between 1 and 30 for MONTHLY_DATE", field)
	case "iso_date_list":
		return fmt.Sprintf("%s must be valid ISO dates (YYYY-MM-DD)", field)
	default:
		return fmt.Sprintf("%s failed validation: %s", field, tag)
	}
}

package migrations

import (
	"context"
	"fmt"
	"log"
	"os"
	"path/filepath"
	"sort"
	"strconv"
	"strings"

	"github.com/jackc/pgx/v5/pgxpool"
)

type Migrator struct {
	pool           *pgxpool.Pool
	migrationsPath string
}

func NewMigrator(pool *pgxpool.Pool, migrationsPath string) *Migrator {
	return &Migrator{
		pool:           pool,
		migrationsPath: migrationsPath,
	}
}

func (m *Migrator) Up(ctx context.Context) error {
	if err := m.ensureMigrationsTable(ctx); err != nil {
		return err
	}

	applied, err := m.getAppliedMigrations(ctx)
	if err != nil {
		return err
	}

	files, err := m.getMigrationFiles(".up.sql")
	if err != nil {
		return err
	}

	for _, file := range files {
		version := m.extractVersion(file)
		if applied[version] {
			continue
		}

		content, err := os.ReadFile(filepath.Join(m.migrationsPath, file))
		if err != nil {
			return fmt.Errorf("reading migration %s: %w", file, err)
		}

		tx, err := m.pool.Begin(ctx)
		if err != nil {
			return fmt.Errorf("starting transaction: %w", err)
		}

		if _, err := tx.Exec(ctx, string(content)); err != nil {
			tx.Rollback(ctx)
			return fmt.Errorf("executing migration %s: %w", file, err)
		}

		if _, err := tx.Exec(ctx, "INSERT INTO schema_migrations (version) VALUES ($1)", version); err != nil {
			tx.Rollback(ctx)
			return fmt.Errorf("recording migration %s: %w", file, err)
		}

		if err := tx.Commit(ctx); err != nil {
			return fmt.Errorf("committing migration %s: %w", file, err)
		}

		log.Printf("Applied migration: %s", file)
	}

	log.Println("Migrations: completed")
	return nil
}

func (m *Migrator) ensureMigrationsTable(ctx context.Context) error {
	query := `
		CREATE TABLE IF NOT EXISTS schema_migrations (
			version BIGINT PRIMARY KEY,
			applied_at TIMESTAMPTZ DEFAULT NOW()
		)
	`
	_, err := m.pool.Exec(ctx, query)
	if err != nil {
		return fmt.Errorf("creating migrations table: %w", err)
	}
	return nil
}

func (m *Migrator) getAppliedMigrations(ctx context.Context) (map[int64]bool, error) {
	rows, err := m.pool.Query(ctx, "SELECT version FROM schema_migrations")
	if err != nil {
		return nil, fmt.Errorf("querying applied migrations: %w", err)
	}
	defer rows.Close()

	applied := make(map[int64]bool)
	for rows.Next() {
		var version int64
		if err := rows.Scan(&version); err != nil {
			return nil, fmt.Errorf("scanning version: %w", err)
		}
		applied[version] = true
	}
	return applied, rows.Err()
}

func (m *Migrator) getMigrationFiles(suffix string) ([]string, error) {
	entries, err := os.ReadDir(m.migrationsPath)
	if err != nil {
		return nil, fmt.Errorf("reading migrations directory: %w", err)
	}

	var files []string
	for _, entry := range entries {
		if !entry.IsDir() && strings.HasSuffix(entry.Name(), suffix) {
			files = append(files, entry.Name())
		}
	}
	sort.Strings(files)
	return files, nil
}

func (m *Migrator) extractVersion(filename string) int64 {
	parts := strings.Split(filename, "_")
	if len(parts) == 0 {
		return 0
	}
	version, _ := strconv.ParseInt(parts[0], 10, 64)
	return version
}

	"github.com/jackc/pgx/v5/pgxpool"
)

func Open(ctx context.Context, dsn string) (*pgxpool.Pool, error) {
	if dsn == "" {
		return nil, fmt.Errorf("database dsn is empty")
	}

	cfg, err := pgxpool.ParseConfig(dsn)
	if err != nil {
		return nil, err
	}

	pool, err := pgxpool.NewWithConfig(ctx, cfg)
	if err != nil {
		return nil, err
	}

	if err := pool.Ping(ctx); err != nil {
		pool.Close()
		return nil, err
	}

	return pool, nil
}

	`

func (r *Repository) CreateGeneratedTask(ctx context.Context, originalTask *taskdomain.Task) (*taskdomain.Task, error) {

	title,

    status,
    recurrence_id,
    is_recurring, 
    parent_task_id, 
    is_generated,
    created_at,
    updated_at
    ) VALUES ($1, $2, $3, $4, $5, $6, $7, NOW(), NOW())
    RETURNING id, created_at, updated_at
	`

	newTask := &taskdomain.Task{
		Title:        originalTask.Title,
		Description:  originalTask.Description,
		Status:       taskdomain.StatusNew,
		RecurrenceID: originalTask.RecurrenceID,
		IsRecurring:  false,
		ParentTaskID: &originalTask.ID,
		IsGenerated:  true,
	}

import (

	recurrencedomain "example.com/taskservice/internal/domain/recurrence"
	taskdomain "example.com/taskservice/internal/domain/task"

type Scheduler struct {
	recurrenceRepo RecurrenceRepository
	taskRepo       TaskRepository
	calculator     Calculator
	interval       time.Duration
	ticker         *time.Ticker
	stopCh         chan struct{}
	wg             sync.WaitGroup
	logger         Logger
}

type Logger interface {
	Printf(format string, v ...interface{})
	Println(v ...interface{})
}

func NewScheduler(
	recurrenceRepo RecurrenceRepository,
	taskRepo TaskRepository,
	calculator Calculator,
	interval time.Duration,
) *Scheduler {
	return &Scheduler{
		recurrenceRepo: recurrenceRepo,
		taskRepo:       taskRepo,
		calculator:     calculator,
		interval:       interval,
		stopCh:         make(chan struct{}),
		logger:         log.Default(),
	}
}

func (s *Scheduler) SetLogger(logger Logger) {
	s.logger = logger
}

func (s *Scheduler) Start(ctx context.Context) {
	s.logger.Printf("Recurrence scheduler started with interval: %v", s.interval)

	s.ticker = time.NewTicker(s.interval)
	defer s.ticker.Stop()

	s.wg.Add(1)
	go func() {
		defer s.wg.Done()
		for {
			select {
			case <-ctx.Done():
				s.logger.Println("Scheduler context cancelled, stopping...")
				return
			case <-s.stopCh:
				s.logger.Println("Scheduler stop signal received, stopping...")
				return
			case <-s.ticker.C:
				if err := s.processRecurrences(ctx); err != nil {
					s.logger.Printf("Error processing recurrences: %v", err)
				}
			}
		}
	}()
}

func (s *Scheduler) Stop() {
	s.logger.Println("Stopping scheduler...")
	close(s.stopCh)
	s.wg.Wait()
	s.logger.Println("Scheduler stopped")
}

func (s *Scheduler) processRecurrences(ctx context.Context) error {
	now := time.Now().UTC()

	configs, err := s.recurrenceRepo.GetActiveRecurrences(ctx, now)
	if err != nil {
		return err
	}

	if len(configs) == 0 {
		return nil
	}

	s.logger.Printf("Processing %d recurrences", len(configs))

	for _, config := range configs {
		if err := s.processRecurrence(ctx, config); err != nil {
			s.logger.Printf("Error processing recurrence ID %d: %v", config.ID, err)
			continue
		}
	}

	return nil
}

func (s *Scheduler) processRecurrence(ctx context.Context, config *recurrencedomain.RecurrenceConfig) error {
	parentTask, err := s.taskRepo.GetByID(ctx, config.TaskID)
	if err != nil {
		return err
	}

	now := time.Now().UTC()

	generatedTask := &taskdomain.Task{
		Title:        parentTask.Title,
		Description:  parentTask.Description,
		Status:       taskdomain.StatusNew,
		IsRecurring:  false,
		ParentTaskID: &parentTask.ID,
		IsGenerated:  true,
		RecurrenceID: &config.ID,
		CreatedAt:    now,
		UpdatedAt:    now,
	}

	_, err = s.taskRepo.Create(ctx, generatedTask)
	if err != nil {
		return err
	}

	nextOccurrence, err := s.calculator.NextOccurrence(config, config.NextOccurrenceAt)
	if err != nil {
		return err
	}

	if err := s.recurrenceRepo.UpdateNextOccurrence(ctx, config.ID, nextOccurrence); err != nil {
		return err
	}

	s.logger.Printf("Generated task from recurrence ID %d, next occurrence: %v", config.ID, nextOccurrence)

	return nil
}
