# Task Service

Сервис для управления задачами с HTTP API на Go.

Документация:

- `docs/recurrence.md` — периодические задачи (расписания) и генерация
- `docs/project-structure.md` — структура проекта и назначение файлов
- `docs/test-task.md` — текст тестового задания

## Требования

- Go `1.23+`
- Docker и Docker Compose

## Быстрый запуск через Docker Compose

```bash
docker compose up --build
```

После запуска сервис будет доступен по адресу `http://localhost:8080`.

Переменные окружения:

- `HTTP_ADDR` (по умолчанию `:8080`)
- `DATABASE_DSN` (по умолчанию `postgres://postgres:postgres@localhost:5432/taskservice?sslmode=disable`)
- `RECURRENCE_LOOKAHEAD_DAYS` (по умолчанию `30`) — окно генерации периодических задач вперёд

Если `postgres` уже запускался ранее со старой схемой, пересоздай volume:

```bash
docker compose down -v
docker compose up --build
```

Причина в том, что SQL-файлы из `migrations/*.sql` монтируются в `docker-entrypoint-initdb.d` и применяются только при инициализации пустого data volume.

## Swagger

Swagger UI:

```text
http://localhost:8080/swagger/
```

OpenAPI JSON:

```text
http://localhost:8080/swagger/openapi.json
```

В Swagger есть примеры запросов для `POST /api/v1/task-schedules` под каждый тип периодичности.

## API

Базовый префикс API:

```text
/api/v1
```

Основные маршруты:

- `POST /api/v1/tasks`
- `GET /api/v1/tasks`
- `GET /api/v1/tasks/{id}`
- `PUT /api/v1/tasks/{id}`
- `DELETE /api/v1/tasks/{id}`

Маршруты расписаний (периодичность):

- `POST /api/v1/task-schedules`
- `GET /api/v1/task-schedules`
- `DELETE /api/v1/task-schedules/{id}`
- `POST /api/v1/task-schedules/generate`

При создании расписания задачи генерируются автоматически на окно `RECURRENCE_LOOKAHEAD_DAYS` (см. `docs/recurrence.md`).

package main

import (
	"context"
	"fmt"
	"log/slog"
	"net/http"
	"os"
	"os/signal"
	"strconv"
	"syscall"
	"time"

	infrastructurepostgres "example.com/taskservice/internal/infrastructure/postgres"
	postgresrepo "example.com/taskservice/internal/repository/postgres"
	transporthttp "example.com/taskservice/internal/transport/http"
	swaggerdocs "example.com/taskservice/internal/transport/http/docs"
	httphandlers "example.com/taskservice/internal/transport/http/handlers"
	scheduleusecase "example.com/taskservice/internal/usecase/schedule"
	"example.com/taskservice/internal/usecase/task"
)

func main() {
	logger := slog.New(slog.NewTextHandler(os.Stdout, &slog.HandlerOptions{
		Level: slog.LevelInfo,
	}))

	cfg := loadConfig()

	ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	pool, err := infrastructurepostgres.Open(ctx, cfg.DatabaseDSN)
	if err != nil {
		logger.Error("open postgres", "error", err)
		os.Exit(1)
	}
	defer pool.Close()

	taskRepo := postgresrepo.New(pool)
	taskUsecase := task.NewService(taskRepo)
	taskHandler := httphandlers.NewTaskHandler(taskUsecase)

	scheduleRepo := postgresrepo.NewScheduleRepository(pool)
	scheduleUsecase := scheduleusecase.NewService(scheduleRepo, taskRepo, cfg.RecurrenceLookaheadDays)
	scheduleHandler := httphandlers.NewTaskScheduleHandler(scheduleUsecase)

	docsHandler := swaggerdocs.NewHandler()
	router := transporthttp.NewRouter(taskHandler, scheduleHandler, docsHandler)

	server := &http.Server{
		Addr:              cfg.HTTPAddr,
		Handler:           router,
		ReadHeaderTimeout: 5 * time.Second,
	}

	go func() {
		<-ctx.Done()

		shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
		defer cancel()

		if err := server.Shutdown(shutdownCtx); err != nil {
			logger.Error("shutdown http server", "error", err)
		}
	}()

	logger.Info("http server started", "addr", cfg.HTTPAddr)

	if err := server.ListenAndServe(); err != nil && err != http.ErrServerClosed {
		logger.Error("listen and serve", "error", err)
		os.Exit(1)
	}
}

type config struct {
	HTTPAddr                string
	DatabaseDSN             string
	RecurrenceLookaheadDays int
}

func loadConfig() config {
	cfg := config{
		HTTPAddr:                envOrDefault("HTTP_ADDR", ":8080"),
		DatabaseDSN:             envOrDefault("DATABASE_DSN", "postgres://postgres:postgres@localhost:5432/taskservice?sslmode=disable"),
		RecurrenceLookaheadDays: envIntOrDefault("RECURRENCE_LOOKAHEAD_DAYS", 30),
	}

	if cfg.DatabaseDSN == "" {
		panic(fmt.Errorf("DATABASE_DSN is required"))
	}

	return cfg
}

func envOrDefault(key, fallback string) string {
	if value := os.Getenv(key); value != "" {
		return value
	}

	return fallback
}

func envIntOrDefault(key string, fallback int) int {
	raw := os.Getenv(key)
	if raw == "" {
		return fallback
	}

	value, err := strconv.Atoi(raw)
	if err != nil || value <= 0 {
		return fallback
	}

	return value
}

# Структура проекта и “что где лежит”

Этот репозиторий — небольшой HTTP API сервис на Go для CRUD задач с хранением в PostgreSQL. Архитектура близка к “clean architecture”: доменная модель отделена от usecase (бизнес-логики), транспорта (HTTP) и инфраструктуры (БД).

## Быстрая карта директорий

```
cmd/
  api/
    main.go                  # Точка входа (старт приложения)
internal/
  domain/task/               # Домен: сущности и доменные ошибки
  usecase/task/              # Usecase/сервис: бизнес-логика, валидация, порты
  repository/postgres/       # Репозиторий: SQL-запросы к Postgres (pgx)
  infrastructure/postgres/   # Инфраструктура: подключение к Postgres (pool)
  transport/http/            # Транспорт: роутер HTTP
    handlers/                # HTTP handlers + DTO (JSON формы запросов/ответов)
    docs/                    # Swagger UI + embedded OpenAPI json
migrations/
  0001_create_tasks.up.sql   # Создание таблицы tasks
Dockerfile                   # Сборка бинарника и runtime image
docker-compose.yml           # Postgres + app для локального запуска
go.mod                       # Зависимости (модули) и версии
go.sum                       # Контрольные суммы модулей (аналог lock+hashes)
README.md                    # Как запускать и куда ходить
```

## Файлы по ключевым зонам

### `cmd/api/main.go` — точка входа

Что делает:
- читает конфиг из env (`HTTP_ADDR`, `DATABASE_DSN`);
- открывает пул к Postgres (`internal/infrastructure/postgres.Open`);
- собирает зависимости (repo → service → handler → router);
- запускает HTTP сервер и корректно завершает по сигналам SIGINT/SIGTERM.

Аналогия с Python:
- примерно как `if __name__ == "__main__":` + создание приложения (`FastAPI()`/`Flask()`), подключение БД, старт сервера.

### `internal/transport/http/router.go` — маршрутизация

- Используется `github.com/gorilla/mux`.
- Все API маршруты под префиксом `/api/v1`.
- Swagger живёт на `/swagger/`, спецификация — `/swagger/openapi.json`.

### `internal/transport/http/handlers/*` — обработчики HTTP

`task_handler.go`:
- `decodeJSON` читает JSON и запрещает неизвестные поля (`DisallowUnknownFields`), это помогает ловить опечатки в запросе.
- `writeUsecaseError` мапит ошибки usecase/domain в HTTP коды.

`dto.go`:
- DTO (структуры для JSON) отделены от доменной модели: доменная `Task` живёт в `internal/domain/task`.

Аналогия с Python:
- DTO ≈ Pydantic models / DRF serializers (форма входа/выхода), отделённые от внутренней модели.

### `internal/usecase/task/*` — бизнес-логика

`service.go`:
- нормализация и валидация (trim строк, дефолтные значения статуса);
- установка `CreatedAt/UpdatedAt` (UTC);
- вызовы репозитория.

`ports.go`:
- интерфейсы (порты) — какие методы должен реализовать репозиторий.

`errors.go`:
- общие ошибки usecase (например, `ErrInvalidInput`).

Аналогия с Python:
- это похоже на слой “services”/“use cases”, куда выносится логика, чтобы handler’ы были тонкими.

### `internal/domain/task/*` — доменная модель

`task.go`:
- `Task` — сущность задачи;
- `Status` — тип статуса + `Valid()` для проверки допустимых значений.

`errors.go`:
- доменные ошибки (например, “не найдено”), которые потом мапятся на HTTP.

### `internal/repository/postgres/*` — доступ к Postgres

`task_repository.go`:
- чистый SQL (INSERT/SELECT/UPDATE/DELETE);
- сканирование результата в доменную `Task`.

Аналогия с Python:
- это похоже на ручной SQL через `psycopg`/`asyncpg`, без ORM.

### `internal/infrastructure/postgres/pool.go` — подключение к БД

- Парсит DSN, создаёт `pgxpool`, делает `Ping`.
- Возвращает готовый `*pgxpool.Pool` для репозиториев.

# Periodic Task Schedules (Recurrence)

В проект добавлена поддержка расписаний (периодичности) для задач. Расписание — это **правило**, по которому сервис может создавать **конкретные задачи-инстансы** на конкретные даты.

Здесь намеренно не реализованы уведомления/напоминания: проверка корректности фичи происходит через API (Swagger/curl) — по факту создания задач с нужными датами.

## Термины

- **Расписание / schedule** — сущность с параметрами периодичности.
- **Задача / task** — конкретная запись в таблице `tasks`. Для периодических задач дополнительно выставляются:
  - `schedule_id` — ссылка на расписание
  - `scheduled_for` — дата (UTC, без времени), на которую создана задача

## Важные допущения

- Все расчёты дат выполняются в **UTC**.
- `scheduled_for` — это именно **дата** (формат `YYYY-MM-DD`), время не учитывается.
- Для ежемесячных задач, если выбранного числа нет в месяце (например, 30 в феврале), задача создаётся **на последний день месяца**.
- Редактирование расписаний и частичное изменение `fixed_dates` не реализованы (для MVP). Если нужно изменить даты — удалите расписание и создайте заново.
- При удалении расписания удаляются:
  - **будущие** задачи (`scheduled_for > today_utc`)
  - задачи на **сегодня**, если их статус `new`
  
  Задачи на сегодня со статусом `in_progress/done`, а также задачи в прошлом остаются как история.

## Возможные улучшения

- Идемпотентность для `POST /api/v1/task-schedules` через заголовок `Idempotency-Key`, чтобы повторный запрос (двойной клик, ретраи клиента) не создавал дубликаты расписаний.
- Транзакции для операций создания/удаления расписания (создание расписания + генерация задач; soft-delete расписания + удаление задач) для более строгой консистентности данных при ошибках.

## Окно генерации

Для генерации используется “окно вперёд”:

- по умолчанию: 30 дней
- настраивается через переменную окружения `RECURRENCE_LOOKAHEAD_DAYS`

Генерация создаёт задачи на диапазон `today_utc .. today_utc + lookahead` (включительно).

## API

Swagger: `http://localhost:8080/swagger/`

В Swagger для `POST /api/v1/task-schedules` есть примеры запросов для всех поддерживаемых типов расписаний.

### Создать расписание

`POST /api/v1/task-schedules`

При создании расписания сервис автоматически генерирует задачи на окно вперёд (см. раздел “Окно генерации”).

Пример: ежедневное (каждый 2-й день)

```json
{
  "title": "Обзвон пациентов",
  "description": "Ежедневные задачи колл-центра",
  "status": "new",
  "type": "daily",
  "payload": { "interval_days": 2, "start_date": "2026-04-13" }
}
```

Типы расписаний:

1) `daily` — каждый `interval_days` день, начиная с `start_date`
```json
{ "interval_days": 1, "start_date": "2026-04-13" }
```

2) `monthly_day` — на `day` каждого месяца (1..30), начиная с `start_date`
```json
{ "day": 30, "start_date": "2026-04-13" }
```


package postgres

import (
	"context"
	"fmt"

	"github.com/jackc/pgx/v5/pgxpool"
)

func Open(ctx context.Context, dsn string) (*pgxpool.Pool, error) {
	if dsn == "" {
		return nil, fmt.Errorf("database dsn is empty")
	}

	cfg, err := pgxpool.ParseConfig(dsn)
	if err != nil {
		return nil, err
	}

	pool, err := pgxpool.NewWithConfig(ctx, cfg)
	if err != nil {
		return nil, err
	}

	if err := pool.Ping(ctx); err != nil {
		pool.Close()
		return nil, err
	}

	return pool, nil
}

	scheduledomain "example.com/taskservice/internal/domain/schedule"
	taskdomain "example.com/taskservice/internal/domain/task"

type Service struct {
	schedules     ScheduleRepository
	tasks         TaskRepository
	lookaheadDays int
	now           func() time.Time
}

func NewService(schedules ScheduleRepository, tasks TaskRepository, lookaheadDays int) *Service {
	if lookaheadDays <= 0 {
		lookaheadDays = 30
	}

	return &Service{
		schedules:     schedules,
		tasks:         tasks,
		lookaheadDays: lookaheadDays,
		now:           func() time.Time { return time.Now().UTC() },
	}
}

type CreateInput struct {
	Title       string
	Description string
	Status      taskdomain.Status
	Type        scheduledomain.Type
	Payload     json.RawMessage
}

func (s *Service) Create(ctx context.Context, input CreateInput) (*scheduledomain.Schedule, error) {
	normalized, err := validateCreateInput(input, utcDate(s.now()))
	if err != nil {
		return nil, err
	}

	now := s.now()
	model := &scheduledomain.Schedule{
		Title:       normalized.Title,
		Description: normalized.Description,
		Status:      normalized.Status,
		Type:        normalized.Type,
		Payload:     normalized.Payload,
		Active:      true,
		CreatedAt:   now,
		UpdatedAt:   now,
	}

	created, err := s.schedules.Create(ctx, model)
	if err != nil {
		return nil, err
	}

	start := utcDate(now)
	end := start.AddDate(0, 0, s.lookaheadDays)
	if _, err := s.generateForSchedule(ctx, *created, start, end, now); err != nil {
		_ = s.schedules.SoftDelete(ctx, created.ID, now)
		return nil, err
	}

	return created, nil
}

func (s *Service) List(ctx context.Context) ([]scheduledomain.Schedule, error) {
	return s.schedules.List(ctx)
}

func (s *Service) Delete(ctx context.Context, id int64) error {
	if id <= 0 {
		return fmt.Errorf("%w: id must be positive", ErrInvalidInput)
	}

	now := s.now()
	today := utcDate(now)

	if err := s.schedules.SoftDelete(ctx, id, now); err != nil {
		return err
	}

	_, err := s.tasks.DeletePlannedByScheduleID(ctx, id, today)
	return err
}

func (s *Service) Generate(ctx context.Context) (int64, error) {
	now := s.now()
	start := utcDate(now)
	end := start.AddDate(0, 0, s.lookaheadDays)

	active, err := s.schedules.ListActive(ctx)
	if err != nil {
		return 0, err
	}

	var generated int64

	for i := range active {
		schedule := active[i]

		count, err := s.generateForSchedule(ctx, schedule, start, end, now)
		if err != nil {
			return 0, err
		}
		generated += count
	}

	return generated, nil
}

func (s *Service) generateForSchedule(ctx context.Context, schedule scheduledomain.Schedule, windowStart, windowEnd, now time.Time) (int64, error) {
	dates, err := datesForSchedule(schedule, windowStart, windowEnd)
	if err != nil {
		return 0, err
	}

	var generated int64
	for _, d := range dates {
		date := d
		scheduleID := schedule.ID

		task := &taskdomain.Task{
			Title:        schedule.Title,
			Description:  schedule.Description,
			Status:       schedule.Status,
			ScheduleID:   &scheduleID,
			ScheduledFor: &date,
			CreatedAt:    now,
			UpdatedAt:    now,
		}

		inserted, err := s.tasks.CreateScheduled(ctx, task)
		if err != nil {
			return 0, err
		}
		if inserted {
			generated++
		}
	}

	return generated, nil
}

func validateCreateInput(input CreateInput, today time.Time) (CreateInput, error) {
	input.Title = strings.TrimSpace(input.Title)
	input.Description = strings.TrimSpace(input.Description)

	if input.Title == "" {
		return CreateInput{}, fmt.Errorf("%w: title is required", ErrInvalidInput)

	if input.Status == "" {
		input.Status = taskdomain.StatusNew

	if !input.Status.Valid() {

	if len(input.Payload) == 0 {

	}
	input.Payload = normalizedPayload

	return input, nil

func normalizePayload(t scheduledomain.Type, raw json.RawMessage, today time.Time) (json.RawMessage, error) {
	switch t {
	case scheduledomain.TypeDaily:
		var p dailyPayload
		if err := json.Unmarshal(raw, &p); err != nil {
			return nil, fmt.Errorf("%w: invalid payload", ErrInvalidInput)
		}
		if p.IntervalDays <= 0 {
			return nil, fmt.Errorf("%w: interval_days must be positive", ErrInvalidInput)
		}
		if p.StartDate == "" {
			p.StartDate = formatDate(today)
		}
		if _, err := parseDate(p.StartDate); err != nil {
			return nil, fmt.Errorf("%w: invalid start_date", ErrInvalidInput)
		}
		return json.Marshal(p)

	case scheduledomain.TypeMonthlyDay:
		var p monthlyDayPayload
		if err := json.Unmarshal(raw, &p); err != nil {
			return nil, fmt.Errorf("%w: invalid payload", ErrInvalidInput)
		}
		if p.Day < 1 || p.Day > 30 {
			return nil, fmt.Errorf("%w: day must be between 1 and 30", ErrInvalidInput)
		}
		if p.StartDate == "" {
			p.StartDate = formatDate(today)
		}
		if _, err := parseDate(p.StartDate); err != nil {
			return nil, fmt.Errorf("%w: invalid start_date", ErrInvalidInput)
		}
		return json.Marshal(p)

	case scheduledomain.TypeFixedDates:
		var p fixedDatesPayload
		if err := json.Unmarshal(raw, &p); err != nil {
			return nil, fmt.Errorf("%w: invalid payload", ErrInvalidInput)
		}
		if len(p.Dates) == 0 {
			return nil, fmt.Errorf("%w: dates is required", ErrInvalidInput)
		}

		seen := make(map[string]struct{}, len(p.Dates))
		normalized := make([]string, 0, len(p.Dates))
		for _, d := range p.Dates {
			d = strings.TrimSpace(d)
			parsed, err := parseDate(d)
			if err != nil {
				return nil, fmt.Errorf("%w: invalid date in dates", ErrInvalidInput)
			}
			key := formatDate(parsed)
			if _, ok := seen[key]; ok {
				continue
			}
			seen[key] = struct{}{}
			normalized = append(normalized, key)
		}
		sort.Strings(normalized)
		p.Dates = normalized
		return json.Marshal(p)

	case scheduledomain.TypeMonthParity:
		var p monthParityPayload
		if err := json.Unmarshal(raw, &p); err != nil {
			return nil, fmt.Errorf("%w: invalid payload", ErrInvalidInput)
		}
		p.Parity = strings.ToLower(strings.TrimSpace(p.Parity))
		if p.Parity != "even" && p.Parity != "odd" {
			return nil, fmt.Errorf("%w: parity must be 'even' or 'odd'", ErrInvalidInput)
		}
		if p.StartDate == "" {
			p.StartDate = formatDate(today)
		}
		if _, err := parseDate(p.StartDate); err != nil {
			return nil, fmt.Errorf("%w: invalid start_date", ErrInvalidInput)
		}
		return json.Marshal(p)

	default:
		return nil, fmt.Errorf("%w: invalid schedule type", ErrInvalidInput)
	}
}

func datesForSchedule(schedule scheduledomain.Schedule, windowStart, windowEnd time.Time) ([]time.Time, error) {
	switch schedule.Type {
	case scheduledomain.TypeDaily:
		var p dailyPayload
		if err := json.Unmarshal(schedule.Payload, &p); err != nil {
			return nil, err
		}
		anchor, err := parseDate(p.StartDate)
		if err != nil {
			return nil, err
		}
		start := maxDate(windowStart, anchor)
		if start.After(windowEnd) {
			return nil, nil
		}
		diff := daysBetween(anchor, start)
		rem := diff % p.IntervalDays
		if rem != 0 {
			start = start.AddDate(0, 0, p.IntervalDays-rem)
		}
		dates := make([]time.Time, 0)
		for d := start; !d.After(windowEnd); d = d.AddDate(0, 0, p.IntervalDays) {
			dates = append(dates, d)
		}
		return dates, nil

	case scheduledomain.TypeMonthlyDay:
		var p monthlyDayPayload
		if err := json.Unmarshal(schedule.Payload, &p); err != nil {
			return nil, err
		}
		anchor, err := parseDate(p.StartDate)
		if err != nil {
			return nil, err
		}
		start := maxDate(windowStart, anchor)
		if start.After(windowEnd) {
			return nil, nil
		}

		cursor := time.Date(start.Year(), start.Month(), 1, 0, 0, 0, 0, time.UTC)
		endMonth := time.Date(windowEnd.Year(), windowEnd.Month(), 1, 0, 0, 0, 0, time.UTC)

		dates := make([]time.Time, 0)
		for !cursor.After(endMonth) {
			lastDay := lastDayOfMonth(cursor.Year(), cursor.Month())
			day := p.Day
			if day > lastDay {
				day = lastDay
			}
			d := time.Date(cursor.Year(), cursor.Month(), day, 0, 0, 0, 0, time.UTC)
			if !d.Before(start) && !d.After(windowEnd) {
				dates = append(dates, d)
			}
			cursor = cursor.AddDate(0, 1, 0)
		}
		return dates, nil

	case scheduledomain.TypeFixedDates:
		var p fixedDatesPayload
		if err := json.Unmarshal(schedule.Payload, &p); err != nil {
			return nil, err
		}
		dates := make([]time.Time, 0, len(p.Dates))
		for _, raw := range p.Dates {
			d, err := parseDate(raw)
			if err != nil {
				return nil, err
			}
			if !d.Before(windowStart) && !d.After(windowEnd) {
				dates = append(dates, d)
			}
		}
		sort.Slice(dates, func(i, j int) bool { return dates[i].Before(dates[j]) })
		return dates, nil

	case scheduledomain.TypeMonthParity:
		var p monthParityPayload
		if err := json.Unmarshal(schedule.Payload, &p); err != nil {
			return nil, err
		}
		anchor, err := parseDate(p.StartDate)
		if err != nil {
			return nil, err
		}
		start := maxDate(windowStart, anchor)
		if start.After(windowEnd) {
			return nil, nil
		}

		wantEven := strings.ToLower(p.Parity) == "even"
		dates := make([]time.Time, 0)
		for d := start; !d.After(windowEnd); d = d.AddDate(0, 0, 1) {
			isEven := d.Day()%2 == 0
			if wantEven == isEven {
				dates = append(dates, d)
			}
		}
		return dates, nil

	default:
		return nil, fmt.Errorf("%w: invalid schedule type", ErrInvalidInput)
	}
}

func utcDate(t time.Time) time.Time {
	t = t.UTC()
	return time.Date(t.Year(), t.Month(), t.Day(), 0, 0, 0, 0, time.UTC)
}

func parseDate(value string) (time.Time, error) {
	parsed, err := time.ParseInLocation("2006-01-02", value, time.UTC)
	if err != nil {
		return time.Time{}, err
	}
	return utcDate(parsed), nil
}

func formatDate(t time.Time) string {
	return utcDate(t).Format("2006-01-02")
}

func daysBetween(from, to time.Time) int {
	from = utcDate(from)
	to = utcDate(to)
	return int(to.Sub(from) / (24 * time.Hour))
}

func maxDate(a, b time.Time) time.Time {
	if a.Before(b) {
		return b
	}
	return a
}

func lastDayOfMonth(year int, month time.Month) int {
	return time.Date(year, month+1, 0, 0, 0, 0, 0, time.UTC).Day()
}
