# Task Service

Сервис для управления задачами и шаблонами периодических задач с HTTP API на Go.

## Требования

- Go `1.23+`
- Docker и Docker Compose

## Быстрый запуск через Docker Compose

```bash
docker compose up --build
```

После запуска сервис будет доступен по адресу `http://localhost:8080`.

Если `postgres` уже запускался ранее со старой схемой, пересоздайте volume:

```bash
docker compose down -v
docker compose up --build
```

Каталог `migrations/` монтируется в `docker-entrypoint-initdb.d`, поэтому SQL-файлы применяются только при инициализации пустого data volume.

## Конфигурация

- `HTTP_ADDR` — адрес HTTP-сервера, по умолчанию `:8080`
- `DATABASE_DSN` — строка подключения к PostgreSQL
- `BUSINESS_TZ` — бизнес-таймзона для расчета календарной даты, по умолчанию `UTC`

## Swagger

Swagger UI:

```text
http://localhost:8080/swagger/
```

OpenAPI JSON:

```text
http://localhost:8080/swagger/openapi.json
```

## API

Базовый префикс API:

```text
/api/v1
```

Маршруты задач:

- `POST /api/v1/tasks`
- `GET /api/v1/tasks`
- `GET /api/v1/tasks/{id}`
- `PUT /api/v1/tasks/{id}`
- `DELETE /api/v1/tasks/{id}`

Маршруты шаблонов периодических задач:

- `POST /api/v1/task-templates`
- `GET /api/v1/task-templates`
- `GET /api/v1/task-templates/{id}`
- `PUT /api/v1/task-templates/{id}`
- `DELETE /api/v1/task-templates/{id}`

## Что реализовано

- Разовые задачи по-прежнему создаются через `/tasks`.
- Периодические настройки вынесены в отдельную сущность `task template`.
- Поддержаны типы периодичности:
  - каждый `n`-й день
  - на конкретное число месяца от `1` до `31`
  - на список конкретных дат
  - на чётные или нечётные дни месяца
- Экземпляры задач материализуются:
  - при создании шаблона
  - при обновлении шаблона
  - при удалении шаблона
  - при старте приложения и затем фоновым проходом раз в минуту

## Допущения

- Шаблон не хранит `status`; все автоматически созданные экземпляры стартуют со статусом `new`.
- Изменение шаблона влияет только на будущие генерации; уже созданные задачи не меняются.
- `DELETE /task-templates/{id}` делает мягкое удаление через `is_active = false`.
- Для ручной задачи `due_date` равен текущей бизнес-дате в `BUSINESS_TZ`.
- Генерация не выполняется ретроактивно раньше даты создания шаблона: `starts_on` нельзя передать раньше текущей бизнес-даты.
- Для режима `every_n_days` отсчет ведется от явного `starts_on`.
- Для режима `day_of_month` значение `31` не поддерживается по условию задания. Если в месяце нет выбранного числа от `1` до `30`, задача создается в последний день этого месяца.

## Тесты

Локально:


Интеграционные тесты репозитория требуют PostgreSQL и переменную окружения `TEST_DATABASE_DSN`. Без неё эти тесты будут пропущены.

package main

import (
	"context"
	"fmt"
	"log/slog"
	"net/http"
	"os"
	"os/signal"
	"syscall"
	"time"

	infrastructurepostgres "example.com/taskservice/internal/infrastructure/postgres"
	postgresrepo "example.com/taskservice/internal/repository/postgres"
	transporthttp "example.com/taskservice/internal/transport/http"
	swaggerdocs "example.com/taskservice/internal/transport/http/docs"
	httphandlers "example.com/taskservice/internal/transport/http/handlers"
	"example.com/taskservice/internal/usecase/task"
	tasktemplate "example.com/taskservice/internal/usecase/tasktemplate"
)

func main() {
	logger := slog.New(slog.NewTextHandler(os.Stdout, &slog.HandlerOptions{
		Level: slog.LevelInfo,
	}))

	cfg := loadConfig()
	location, err := time.LoadLocation(cfg.BusinessTZ)
	if err != nil {
		logger.Error("load business timezone", "timezone", cfg.BusinessTZ, "error", err)
		os.Exit(1)
	}

	ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	pool, err := infrastructurepostgres.Open(ctx, cfg.DatabaseDSN)
	if err != nil {
		logger.Error("open postgres", "error", err)
		os.Exit(1)
	}
	defer pool.Close()

	taskRepo := postgresrepo.New(pool)
	taskTemplateRepo := postgresrepo.NewTaskTemplateRepository(pool)
	taskUsecase := task.NewService(taskRepo, location)
	taskTemplateUsecase := tasktemplate.NewService(taskTemplateRepo, location)
	taskHandler := httphandlers.NewTaskHandler(taskUsecase)
	taskTemplateHandler := httphandlers.NewTaskTemplateHandler(taskTemplateUsecase)
	docsHandler := swaggerdocs.NewHandler()
	router := transporthttp.NewRouter(taskHandler, taskTemplateHandler, docsHandler)

	server := &http.Server{
		Addr:              cfg.HTTPAddr,
		Handler:           router,
		ReadHeaderTimeout: 5 * time.Second,
	}

	go func() {
		<-ctx.Done()

		shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
		defer cancel()

		if err := server.Shutdown(shutdownCtx); err != nil {
			logger.Error("shutdown http server", "error", err)
		}
	}()

	go func() {
		if err := taskTemplateUsecase.MaterializeDueTemplates(ctx); err != nil {
			logger.Error("materialize task templates on start", "error", err)
		}

		ticker := time.NewTicker(time.Minute)
		defer ticker.Stop()

		for {
			select {
			case <-ctx.Done():
				return
			case <-ticker.C:
				if err := taskTemplateUsecase.MaterializeDueTemplates(ctx); err != nil {
					logger.Error("materialize task templates", "error", err)
				}
			}
		}
	}()

	logger.Info("http server started", "addr", cfg.HTTPAddr)

	if err := server.ListenAndServe(); err != nil && err != http.ErrServerClosed {
		logger.Error("listen and serve", "error", err)
		os.Exit(1)
	}
}

type config struct {
	HTTPAddr    string
	DatabaseDSN string
	BusinessTZ  string
}

func loadConfig() config {
	cfg := config{
		HTTPAddr:    envOrDefault("HTTP_ADDR", ":8080"),
		DatabaseDSN: envOrDefault("DATABASE_DSN", "postgres://postgres:postgres@localhost:5432/taskservice?sslmode=disable"),
		BusinessTZ:  envOrDefault("BUSINESS_TZ", "UTC"),
	}

	if cfg.DatabaseDSN == "" {
		panic(fmt.Errorf("DATABASE_DSN is required"))
	}

	return cfg
}

func envOrDefault(key, fallback string) string {
	if value := os.Getenv(key); value != "" {
		return value
	}

	return fallback
}

package calendar

import "time"

const DateLayout = "2006-01-02"

func BusinessDate(now time.Time, location *time.Location) time.Time {
	year, month, day := now.In(location).Date()

	return time.Date(year, month, day, 0, 0, 0, 0, time.UTC)
}

func NormalizeDate(value time.Time) time.Time {
	year, month, day := value.UTC().Date()

	return time.Date(year, month, day, 0, 0, 0, 0, time.UTC)
}

func ParseDate(value string) (time.Time, error) {
	parsed, err := time.Parse(DateLayout, value)
	if err != nil {
		return time.Time{}, err
	}

	return NormalizeDate(parsed), nil
}

func FormatDate(value time.Time) string {
	if value.IsZero() {
		return ""
	}

	return NormalizeDate(value).Format(DateLayout)
}

func DaysBetween(start, end time.Time) int {
	start = NormalizeDate(start)
	end = NormalizeDate(end)

	return int(end.Sub(start).Hours() / 24)
}

func DatePtr(value time.Time) *time.Time {
	normalized := NormalizeDate(value)
	return &normalized
}

package tasktemplate

import (
	"time"

	"example.com/taskservice/internal/calendar"
)

type RecurrenceType string

const (
	RecurrenceEveryNDays  RecurrenceType = "every_n_days"
	RecurrenceDayOfMonth  RecurrenceType = "day_of_month"
	RecurrenceSpecific    RecurrenceType = "specific_dates"
	RecurrenceDayParity   RecurrenceType = "day_parity"
)

type DayParity string

const (
	DayParityOdd  DayParity = "odd"
	DayParityEven DayParity = "even"
)

type Recurrence struct {
	Type          RecurrenceType `json:"type"`
	StartsOn      time.Time      `json:"starts_on"`
	EveryNDays    *int           `json:"every_n_days,omitempty"`
	DayOfMonth    *int           `json:"day_of_month,omitempty"`
	SpecificDates []time.Time    `json:"specific_dates,omitempty"`
	DayParity     *DayParity     `json:"parity,omitempty"`
}

type TaskTemplate struct {
	ID                  int64      `json:"id"`
	Title               string     `json:"title"`
	Description         string     `json:"description"`
	IsActive            bool       `json:"is_active"`
	Recurrence          Recurrence `json:"recurrence"`
	LastMaterializedFor *time.Time `json:"-"`
	CreatedAt           time.Time  `json:"created_at"`
	UpdatedAt           time.Time  `json:"updated_at"`
}

func (t RecurrenceType) Valid() bool {
	switch t {
	case RecurrenceEveryNDays, RecurrenceDayOfMonth, RecurrenceSpecific, RecurrenceDayParity:
		return true
	default:
		return false
	}
}

func (p DayParity) Valid() bool {
	switch p {
	case DayParityOdd, DayParityEven:
		return true
	default:
		return false
	}
}

func (r Recurrence) Matches(date time.Time) bool {
	date = calendar.NormalizeDate(date)
	startsOn := calendar.NormalizeDate(r.StartsOn)

	if date.Before(startsOn) {
		return false
	}

	switch r.Type {
	case RecurrenceEveryNDays:
		if r.EveryNDays == nil || *r.EveryNDays <= 0 {
			return false
		}

		return calendar.DaysBetween(startsOn, date)%*r.EveryNDays == 0
	case RecurrenceDayOfMonth:
		if r.DayOfMonth == nil {
			return false
		}

		targetDay := *r.DayOfMonth
		lastDayOfMonth := daysInMonth(date.Year(), date.Month())
		if targetDay > lastDayOfMonth {
			targetDay = lastDayOfMonth
		}

		return date.Day() == targetDay
	case RecurrenceSpecific:
		for _, specificDate := range r.SpecificDates {
			if calendar.NormalizeDate(specificDate).Equal(date) {
				return true
			}
		}

		return false
	case RecurrenceDayParity:
		if r.DayParity == nil {
			return false
		}

		if *r.DayParity == DayParityEven {
			return date.Day()%2 == 0
		}

		return date.Day()%2 != 0
	default:
		return false
	}
}

func daysInMonth(year int, month time.Month) int {
	return time.Date(year, month+1, 0, 0, 0, 0, 0, time.UTC).Day()
}

func Open(ctx context.Context, dsn string) (*pgxpool.Pool, error) {
	if dsn == "" {
		return nil, fmt.Errorf("database dsn is empty")
	}

	cfg, err := pgxpool.ParseConfig(dsn)
	if err != nil {
		return nil, err
	}

	pool, err := pgxpool.NewWithConfig(ctx, cfg)
	if err != nil {
		return nil, err
	}

	if err := pool.Ping(ctx); err != nil {
		pool.Close()
		return nil, err
	}

	return pool, nil
}

func (r *TaskTemplateRepository) Materialize(
	ctx context.Context,
	template *tasktemplatedomain.TaskTemplate,
	upTo time.Time,
	createdAt time.Time,
) error {
	upTo = calendar.NormalizeDate(upTo)
	createdAt = createdAt.UTC()

	tx, err := r.pool.BeginTx(ctx, pgx.TxOptions{})
	if err != nil {
		return err
	}
	defer func() {
		_ = tx.Rollback(ctx)
	}()

	start := template.Recurrence.StartsOn
	if template.LastMaterializedFor != nil {
		start = calendar.NormalizeDate(template.LastMaterializedFor.AddDate(0, 0, 1))
	}

	if start.Before(template.Recurrence.StartsOn) {
		start = template.Recurrence.StartsOn
	}

	if !start.After(upTo) {
		const insertQuery = `
			INSERT INTO tasks (title, description, status, template_id, due_date, created_at, updated_at)
			VALUES ($1, $2, $3, $4, $5, $6, $6)
			ON CONFLICT (template_id, due_date) WHERE template_id IS NOT NULL DO NOTHING
		`

		for date := start; !date.After(upTo); date = date.AddDate(0, 0, 1) {
			if !template.Recurrence.Matches(date) {
				continue
			}

			_, err = tx.Exec(
				ctx,
				insertQuery,
				template.Title,
				template.Description,
				taskdomain.StatusNew,
				template.ID,
				date,
				createdAt,
			)
			if err != nil {
				return err
			}
		}
	}

	const updateQuery = `
		UPDATE task_templates
		SET last_materialized_for = $1
		WHERE id = $2
	`

	if _, err = tx.Exec(ctx, updateQuery, upTo, template.ID); err != nil {
		return err
	}

	if err = tx.Commit(ctx); err != nil {
		return err
	}

	template.LastMaterializedFor = calendar.DatePtr(upTo)

	return nil
}

func NewService(repo Repository, location *time.Location) *Service {
	if location == nil {
		location = time.UTC
	}

	return &Service{
		repo:     repo,
		now:      func() time.Time { return time.Now().UTC() },

func (s *Service) Create(ctx context.Context, input CreateInput) (*tasktemplatedomain.TaskTemplate, error) {
	now, today := s.currentTimes()

	normalized, err := normalizeInput(input.Title, input.Description, input.IsActive, input.Recurrence, today, true)
	if err != nil {
		return nil, err
	}

	model := &tasktemplatedomain.TaskTemplate{
		Title:       normalized.Title,
		Description: normalized.Description,
		IsActive:    normalized.IsActive,
		Recurrence:  normalized.Recurrence,
		CreatedAt:   now,
		UpdatedAt:   now,
	}

	created, err := s.repo.Create(ctx, model)
	if err != nil {
		return nil, err
	}

	if created.IsActive {
		if err := s.repo.Materialize(ctx, created, today, now); err != nil {
			return nil, err
		}

		created.LastMaterializedFor = calendar.DatePtr(today)
	}

	return created, nil
}

func (s *Service) GetByID(ctx context.Context, id int64) (*tasktemplatedomain.TaskTemplate, error) {
	if id <= 0 {
		return nil, fmt.Errorf("%w: id must be positive", ErrInvalidInput)
	}

	return s.repo.GetByID(ctx, id)
}

func (s *Service) Update(ctx context.Context, id int64, input UpdateInput) (*tasktemplatedomain.TaskTemplate, error) {
	if id <= 0 {
		return nil, fmt.Errorf("%w: id must be positive", ErrInvalidInput)
	}

	now, today := s.currentTimes()

	existing, err := s.repo.GetByID(ctx, id)
	if err != nil {
		return nil, err
	}

	if existing.IsActive {
		if err := s.repo.Materialize(ctx, existing, today, now); err != nil {
			return nil, err
		}
	}

	defaultActive := existing.IsActive
	normalized, err := normalizeInput(input.Title, input.Description, input.IsActive, input.Recurrence, today, defaultActive)
	if err != nil {
		return nil, err
	}

	model := &tasktemplatedomain.TaskTemplate{
		ID:                  id,
		Title:               normalized.Title,
		Description:         normalized.Description,
		IsActive:            normalized.IsActive,
		Recurrence:          normalized.Recurrence,
		LastMaterializedFor: calendar.DatePtr(today),
		CreatedAt:           existing.CreatedAt,
		UpdatedAt:           now,
	}

	updated, err := s.repo.Update(ctx, model)
	if err != nil {
		return nil, err
	}

	return updated, nil
}

func normalizeInput(
	title string,
	description string,
	isActive *bool,
	recurrenceInput RecurrenceInput,
	today time.Time,
	defaultActive bool,
) (normalizedInput, error) {
	title = strings.TrimSpace(title)
	description = strings.TrimSpace(description)

	if title == "" {
		return normalizedInput{}, fmt.Errorf("%w: title is required", ErrInvalidInput)
	}

	recurrence, err := normalizeRecurrence(recurrenceInput, today)
	if err != nil {
		return normalizedInput{}, err
	}

	active := defaultActive
	if isActive != nil {
		active = *isActive
	}

	return normalizedInput{
		Title:       title,
		Description: description,
		IsActive:    active,
		Recurrence:  recurrence,
	}, nil
}

func normalizeRecurrence(input RecurrenceInput, today time.Time) (tasktemplatedomain.Recurrence, error) {
	recurrenceType := tasktemplatedomain.RecurrenceType(strings.TrimSpace(string(input.Type)))
	if !recurrenceType.Valid() {
		return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: invalid recurrence type", ErrInvalidInput)
	}

	startsOn, err := calendar.ParseDate(strings.TrimSpace(input.StartsOn))
	if err != nil {
		return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: invalid starts_on", ErrInvalidInput)
	}

	if startsOn.Before(today) {
		return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: starts_on must not be earlier than today", ErrInvalidInput)
	}

	recurrence := tasktemplatedomain.Recurrence{
		Type:     recurrenceType,
		StartsOn: startsOn,
	}

	switch recurrenceType {
	case tasktemplatedomain.RecurrenceEveryNDays:
		if len(input.SpecificDates) > 0 || input.DayOfMonth != nil || input.DayParity != nil {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: unexpected fields for every_n_days recurrence", ErrInvalidInput)
		}

		if input.EveryNDays == nil || *input.EveryNDays < 1 {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: every_n_days must be greater than zero", ErrInvalidInput)
		}

		value := *input.EveryNDays
		recurrence.EveryNDays = &value
	case tasktemplatedomain.RecurrenceDayOfMonth:
		if len(input.SpecificDates) > 0 || input.EveryNDays != nil || input.DayParity != nil {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: unexpected fields for day_of_month recurrence", ErrInvalidInput)
		}

		if input.DayOfMonth == nil || *input.DayOfMonth < 1 || *input.DayOfMonth > 30 {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: day_of_month must be between 1 and 30", ErrInvalidInput)
		}

		value := *input.DayOfMonth
		recurrence.DayOfMonth = &value
	case tasktemplatedomain.RecurrenceSpecific:
		if input.EveryNDays != nil || input.DayOfMonth != nil || input.DayParity != nil {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: unexpected fields for specific_dates recurrence", ErrInvalidInput)
		}

		if len(input.SpecificDates) == 0 {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: specific_dates must not be empty", ErrInvalidInput)
		}

		normalizedDates, err := normalizeSpecificDates(input.SpecificDates, startsOn)
		if err != nil {
			return tasktemplatedomain.Recurrence{}, err
		}

		recurrence.SpecificDates = normalizedDates
	case tasktemplatedomain.RecurrenceDayParity:
		if len(input.SpecificDates) > 0 || input.EveryNDays != nil || input.DayOfMonth != nil {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: unexpected fields for day_parity recurrence", ErrInvalidInput)
		}

		if input.DayParity == nil {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: parity is required", ErrInvalidInput)
		}

		parity := tasktemplatedomain.DayParity(strings.TrimSpace(string(*input.DayParity)))
		if !parity.Valid() {
			return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: invalid parity", ErrInvalidInput)
		}

		recurrence.DayParity = &parity
	default:
		return tasktemplatedomain.Recurrence{}, fmt.Errorf("%w: unsupported recurrence type", ErrInvalidInput)
	}

	return recurrence, nil
}

func normalizeSpecificDates(values []string, startsOn time.Time) ([]time.Time, error) {
	unique := make(map[string]struct{}, len(values))
	dates := make([]time.Time, 0, len(values))

	for _, rawValue := range values {
		date, err := calendar.ParseDate(strings.TrimSpace(rawValue))
		if err != nil {
			return nil, fmt.Errorf("%w: invalid specific_dates value", ErrInvalidInput)
		}

		if date.Before(startsOn) {
			return nil, fmt.Errorf("%w: specific_dates must not be earlier than starts_on", ErrInvalidInput)
		}

		formatted := calendar.FormatDate(date)
		if _, exists := unique[formatted]; exists {
			return nil, fmt.Errorf("%w: specific_dates must be unique", ErrInvalidInput)
		}

		unique[formatted] = struct{}{}
		dates = append(dates, date)
	}

	sort.Slice(dates, func(i, j int) bool {
		return dates[i].Before(dates[j])
	})

	return dates, nil
}

func (r *fakeRepository) Materialize(
	_ context.Context,
	template *tasktemplatedomain.TaskTemplate,
	upTo time.Time,
	_ time.Time,
) error {
	stored, ok := r.templates[template.ID]
	if !ok {
		return tasktemplatedomain.ErrNotFound
	}

	start := stored.Recurrence.StartsOn
	if stored.LastMaterializedFor != nil {
		start = stored.LastMaterializedFor.AddDate(0, 0, 1)
	}

	if start.Before(stored.Recurrence.StartsOn) {
		start = stored.Recurrence.StartsOn
	}

	if r.generated[stored.ID] == nil {
		r.generated[stored.ID] = make(map[string]struct{})
	}

	for date := start; !date.After(upTo); date = date.AddDate(0, 0, 1) {
		if !stored.Recurrence.Matches(date) {

		clone.EveryNDays = &value

	if recurrence.DayOfMonth != nil {
		value := *recurrence.DayOfMonth
		clone.DayOfMonth = &value
	}

	}

}
