# Внесенные изменения

## Данные и миграции

- Добавлена миграция `0002`: таблицы **`schedule`** (типы: каждые N дней, день месяца, чёт/нечёт, фиксированные даты) и **`fixed_date`** для конкретных календарных дат задачи.
- В **docker-compose** каталог `migrations` целиком монтируется в контейнер Postgres, чтобы поднимались все SQL-миграции, не только первая.

## Доменная логика

- У **`Task`** появилась связанная **`Schedule`**.
- Реализован расчёт **следующей даты выполнения** по расписанию (`NextExecutionAfter` и связанные функции), с юнит-тестами.
- Добавлены доменные ошибки, в т.ч. для случая «у задачи нет расписания».

## Слой репозитория (PostgreSQL)

- Репозитории для **`schedule`** и **`fixed_date`**, доработан **`task_repository`**: создание задачи с расписанием, чтение расписания, интеграционные тесты.
- Вспомогательный код для пула/тестов (`pool.go`, `helpers_test.go`).

## Use case

- **Create** поддерживает передачу расписания; после создания расписание подгружается в ответ.
- Новый сценарий **`NextRun`**: по `task id` и опорной дате возвращается следующий календарный день выполнения (UTC).

## HTTP API и документация

- Новый эндпоинт **`GET /api/v1/tasks/{id}/next-run`** (query-параметр опорной даты).
- DTO и **OpenAPI** обновлены под расписание и `next-run`.
- Добавлены тесты хендлеров; целевой интерфейс use case для хендлеров; **mocks** (`go generate`) для тестов.

## Прочее

- Зависимости в **go.mod** / **go.sum** обновлены под новый функционал и тесты.
- В **Makefile** добавлена цель **`mock`** (`go generate ./...`).

# Task Service

Сервис для управления задачами с HTTP API на Go.

## Требования

- Go `1.23+`
- Docker и Docker Compose

## Быстрый запуск через Docker Compose

```bash
docker compose up --build
```

После запуска сервис будет доступен по адресу `http://localhost:8080`.

Если `postgres` уже запускался ранее со старой схемой, пересоздай volume:

```bash
docker compose down -v
docker compose up --build
```

Причина в том, что SQL-файл из `migrations/0001_create_tasks.up.sql` монтируется в `docker-entrypoint-initdb.d` и применяется только при инициализации пустого data volume.

## Swagger

Swagger UI:

```text
http://localhost:8080/swagger/
```

OpenAPI JSON:

```text
http://localhost:8080/swagger/openapi.json
```

## API

Базовый префикс API:

```text
/api/v1
```

Основные маршруты:

- `POST /api/v1/tasks`
- `GET /api/v1/tasks`
- `GET /api/v1/tasks/{id}`
- `PUT /api/v1/tasks/{id}`
- `DELETE /api/v1/tasks/{id}`

package main

import (
	"context"
	"fmt"
	"log/slog"
	"net/http"
	"os"
	"os/signal"
	"syscall"
	"time"

	infrastructurepostgres "example.com/taskservice/internal/infrastructure/postgres"
	postgresrepo "example.com/taskservice/internal/repository/postgres"
	transporthttp "example.com/taskservice/internal/transport/http"
	swaggerdocs "example.com/taskservice/internal/transport/http/docs"
	httphandlers "example.com/taskservice/internal/transport/http/handlers"
	"example.com/taskservice/internal/usecase/task"
)

func main() {
	logger := slog.New(slog.NewTextHandler(os.Stdout, &slog.HandlerOptions{
		Level: slog.LevelInfo,
	}))

	cfg := loadConfig()

	ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	pool, err := infrastructurepostgres.Open(ctx, cfg.DatabaseDSN)
	if err != nil {
		logger.Error("open postgres", "error", err)
		os.Exit(1)
	}
	defer pool.Close()

	taskRepo := postgresrepo.New(pool)
	taskUsecase := task.NewService(taskRepo)
	taskHandler := httphandlers.NewTaskHandler(taskUsecase)
	docsHandler := swaggerdocs.NewHandler()
	router := transporthttp.NewRouter(taskHandler, docsHandler)

	server := &http.Server{
		Addr:              cfg.HTTPAddr,
		Handler:           router,
		ReadHeaderTimeout: 5 * time.Second,
	}

	go func() {
		<-ctx.Done()

		shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
		defer cancel()

		if err := server.Shutdown(shutdownCtx); err != nil {
			logger.Error("shutdown http server", "error", err)
		}
	}()

	logger.Info("http server started", "addr", cfg.HTTPAddr)

	if err := server.ListenAndServe(); err != nil && err != http.ErrServerClosed {
		logger.Error("listen and serve", "error", err)
		os.Exit(1)
	}
}

type config struct {
	HTTPAddr    string
	DatabaseDSN string
}

func loadConfig() config {
	cfg := config{
		HTTPAddr:    envOrDefault("HTTP_ADDR", ":8080"),
		DatabaseDSN: envOrDefault("DATABASE_DSN", "postgres://postgres:postgres@localhost:5432/taskservice?sslmode=disable"),
	}

	if cfg.DatabaseDSN == "" {
		panic(fmt.Errorf("DATABASE_DSN is required"))
	}

	return cfg
}

func envOrDefault(key, fallback string) string {
	if value := os.Getenv(key); value != "" {
		return value
	}

	return fallback
}

package task

import "time"

// Schedule type values (aligned with DB / repository).
const (
	ScheduleTypeAfterNDays = 1 + iota
	ScheduleTypeDayOfMonth
	ScheduleTypeOddOrEven
	ScheduleTypeCertainDates
)

// NextExecutionAfter returns the earliest calendar execution date strictly after ref (UTC date boundaries).
// anchorCreated is the task creation time (used for every-N-days schedules).
func NextExecutionAfter(anchorCreated time.Time, s *Schedule, ref time.Time) (time.Time, bool) {
	if s == nil {
		return time.Time{}, false
	}

	refDay := startOfDayUTC(ref)
	anchorDay := startOfDayUTC(anchorCreated)

	switch s.Type {
	case ScheduleTypeAfterNDays:
		if s.AfterAmountOfDays == nil || *s.AfterAmountOfDays <= 0 {
			return time.Time{}, false
		}
		return nextEveryNDaysAfter(anchorDay, refDay, *s.AfterAmountOfDays), true

	case ScheduleTypeDayOfMonth:
		if s.DayOfMonth == nil {
			return time.Time{}, false
		}
		return nextDayOfMonthAfter(refDay, *s.DayOfMonth)

	case ScheduleTypeOddOrEven:
		if s.OddOrEven == nil {
			return time.Time{}, false
		}
		wantOdd := *s.OddOrEven == "ODD"
		return nextOddEvenCalendarDayAfter(refDay, wantOdd), true

	case ScheduleTypeCertainDates:
		return nextCertainDateAfter(s.CertainDays, refDay)

	default:
		return time.Time{}, false
	}
}

func startOfDayUTC(t time.Time) time.Time {
	t = t.UTC()
	return time.Date(t.Year(), t.Month(), t.Day(), 0, 0, 0, 0, time.UTC)
}

func nextEveryNDaysAfter(anchor, ref time.Time, n int) time.Time {
	d := anchor
	for !d.After(ref) {
		d = d.AddDate(0, 0, n)
	}
	return d
}

func daysInMonth(year int, m time.Month) int {
	return time.Date(year, m+1, 0, 0, 0, 0, 0, time.UTC).Day()
}

func nextDayOfMonthAfter(ref time.Time, dom int) (time.Time, bool) {
	y, mo, _ := ref.Date()
	base := time.Date(y, mo, 1, 0, 0, 0, 0, time.UTC)

	for i := 0; i < 1200; i++ {
		cur := base.AddDate(0, i, 0)
		cy, cm, _ := cur.Date()
		last := daysInMonth(cy, time.Month(cm))
		if dom > last {
			continue
		}
		cand := time.Date(cy, time.Month(cm), dom, 0, 0, 0, 0, time.UTC)
		if cand.After(ref) {
			return cand, true
		}
	}
	return time.Time{}, false
}

// nextOddEvenCalendarDayAfter finds the first calendar day strictly after ref whose day-of-month is odd (ODD) or even (EVEN).
func nextOddEvenCalendarDayAfter(ref time.Time, wantOdd bool) time.Time {
	d := ref.AddDate(0, 0, 1)
	for i := 0; i < 800; i++ {
		day := d.Day()
		isOdd := day%2 == 1
		if wantOdd == isOdd {
			return d
		}
		d = d.AddDate(0, 0, 1)
	}
	return d
}

func nextCertainDateAfter(dates []time.Time, ref time.Time) (time.Time, bool) {
	if len(dates) == 0 {
		return time.Time{}, false
	}

	var best *time.Time
	for i := range dates {
		day := startOfDayUTC(dates[i])
		if !day.After(ref) {
			continue
		}
		if best == nil || day.Before(*best) {
			t := day
			best = &t
		}
	}
	if best == nil {
		return time.Time{}, false
	}
	return *best, true
}

		return nil, err

	}

	return pool, nil
}

func composeInsertDate(taskID int64, dates []time.Time) (string, []any) {
	if len(dates) == 0 {
		return "", nil
	}

	args := make([]any, 0, len(dates)*2)
	b := []byte("INSERT INTO fixed_date (task_id, certain_date) VALUES ")
	n := 1
	for i := range dates {
		if i > 0 {
			b = append(b, []byte(", ")...)
		}
		b = fmt.Appendf(b, "($%d, $%d)", n, n+1)
		n += 2
		args = append(args, taskID, dates[i])
	}

	return string(b), args
}

func (r *Repository) createSchedule(ctx context.Context, e dbExecutor, schedule *taskdomain.Schedule) (*taskdomain.Schedule, error) {
	query, args := composeInsertSchedule(schedule)

	if _, err := e.Exec(ctx, query, args...); err != nil {
		return nil, fmt.Errorf("CreateSchedule - %w", err)
	}

	if schedule.Type == CERTAIN_DATE {
		if err := r.createFixedDates(ctx, e, schedule.TaskID, schedule.CertainDays); err != nil {
			return nil, err
		}
	}

	return schedule, nil

func composeInsertSchedule(s *taskdomain.Schedule) (string, []any) {
	cols := []string{"task_id", "type"}
	args := []any{s.TaskID, s.Type}

	if s.AfterAmountOfDays != nil {
		cols = append(cols, "after_amount_of_days")
		args = append(args, *s.AfterAmountOfDays)
	}
	if s.DayOfMonth != nil {
		cols = append(cols, "day_of_month")
		args = append(args, *s.DayOfMonth)
	}
	if s.OddOrEven != nil {
		cols = append(cols, "odd_or_even")
		args = append(args, *s.OddOrEven)
	}

	ph := make([]string, len(args))
	for i := range args {
		ph[i] = fmt.Sprintf("$%d", i+1)
	}

	q := fmt.Sprintf(
		"INSERT INTO schedule (%s) VALUES (%s)",
		strings.Join(cols, ", "),
		strings.Join(ph, ", "),
	)
	return q, args
}

func (r *Repository) GetSchedule(ctx context.Context, taskID int64) (*taskdomain.Schedule, error) {
	const q = `
		SELECT task_id, type, after_amount_of_days, day_of_month, odd_or_even
		FROM schedule
		WHERE task_id = $1
	`
	row := r.pool.QueryRow(ctx, q, taskID)

	var s taskdomain.Schedule
	var after, dom *int
	var odd *string

	if err := row.Scan(&s.TaskID, &s.Type, &after, &dom, &odd); err != nil {
		if errors.Is(err, pgx.ErrNoRows) {
			return nil, nil
		}

		return nil, err
	}

	s.AfterAmountOfDays = after
	s.DayOfMonth = dom
	s.OddOrEven = odd

	if s.Type == CERTAIN_DATE {
		rows, err := r.pool.Query(ctx, `SELECT certain_date FROM fixed_date WHERE task_id = $1 ORDER BY certain_date`, taskID)
		if err != nil {
			return nil, err
		}
		defer rows.Close()

		for rows.Next() {
			var dt time.Time
			if err := rows.Scan(&dt); err != nil {
				return nil, err
			}
			s.CertainDays = append(s.CertainDays, dt)
		}
		if err := rows.Err(); err != nil {
			return nil, err
		}
	}

	return &s, nil
}

func (r *Repository) replaceScheduleExec(ctx context.Context, e dbExecutor, schedule *taskdomain.Schedule) error {
	if _, err := e.Exec(ctx, `DELETE FROM fixed_date WHERE task_id = $1`, schedule.TaskID); err != nil {
		return fmt.Errorf("ReplaceSchedule delete fixed_date - %w", err)
	}
	if _, err := e.Exec(ctx, `DELETE FROM schedule WHERE task_id = $1`, schedule.TaskID); err != nil {
		return fmt.Errorf("ReplaceSchedule delete schedule - %w", err)
	}

	q, args := composeInsertSchedule(schedule)
	if _, err := e.Exec(ctx, q, args...); err != nil {
		return fmt.Errorf("ReplaceSchedule insert schedule - %w", err)
	}

	if schedule.Type == CERTAIN_DATE {
		q2, args2 := composeInsertDate(schedule.TaskID, schedule.CertainDays)
		if q2 != "" {
			if _, err := e.Exec(ctx, q2, args2...); err != nil {
				return fmt.Errorf("ReplaceSchedule insert fixed_date - %w", err)
			}
		}
	}

	return nil
}

func (r *Repository) ReplaceSchedule(ctx context.Context, schedule *taskdomain.Schedule) error {
	tx, err := r.pool.Begin(ctx)
	if err != nil {
		return fmt.Errorf("ReplaceSchedule begin - %w", err)
	}
	defer func() { _ = tx.Rollback(ctx) }()

	if err := r.replaceScheduleExec(ctx, tx, schedule); err != nil {
		return err
	}

	if err := tx.Commit(ctx); err != nil {
		return fmt.Errorf("ReplaceSchedule commit - %w", err)
	}

	return nil
}

func (r *Repository) create(ctx context.Context, e dbExecutor, task *taskdomain.Task) (*taskdomain.Task, error) {

// CreateWithSchedule inserts the task and schedule (and fixed dates when applicable) in one transaction.
func (r *Repository) CreateWithSchedule(ctx context.Context, task *taskdomain.Task, schedule *taskdomain.Schedule) (*taskdomain.Task, error) {
	tx, err := r.pool.Begin(ctx)
	if err != nil {
		return nil, fmt.Errorf("CreateWithSchedule begin - %w", err)
	}
	defer func() { _ = tx.Rollback(ctx) }()

	created, err := r.create(ctx, tx, task)
	if err != nil {
		return nil, err
	}

	schedule.TaskID = created.ID
	if _, err := r.createSchedule(ctx, tx, schedule); err != nil {
		return nil, err
	}

	if err := tx.Commit(ctx); err != nil {
		return nil, fmt.Errorf("CreateWithSchedule commit - %w", err)
	}

	return created, nil
}

// UpdateWithReplaceSchedule updates the task and replaces its schedule in one transaction.
func (r *Repository) UpdateWithReplaceSchedule(ctx context.Context, task *taskdomain.Task, schedule *taskdomain.Schedule) (*taskdomain.Task, error) {
	tx, err := r.pool.Begin(ctx)
	if err != nil {
		return nil, fmt.Errorf("UpdateWithReplaceSchedule begin - %w", err)

	defer func() { _ = tx.Rollback(ctx) }()

	updated, err := r.update(ctx, tx, task)
	if err != nil {
		return nil, err
	}

	if err := r.replaceScheduleExec(ctx, tx, schedule); err != nil {

// UpdateAndRemoveSchedule обновляет поля задачи и удаляет строки schedule/fixed_date в одной транзакции.
func (r *Repository) UpdateAndRemoveSchedule(ctx context.Context, task *taskdomain.Task) (*taskdomain.Task, error) {
	tx, err := r.pool.Begin(ctx)
	if err != nil {
		return nil, fmt.Errorf("UpdateAndRemoveSchedule begin - %w", err)
	}
	defer func() { _ = tx.Rollback(ctx) }()

	updated, err := r.update(ctx, tx, task)
	if err != nil {
		return nil, err
	}

	if _, err := tx.Exec(ctx, `DELETE FROM fixed_date WHERE task_id = $1`, task.ID); err != nil {
		return nil, fmt.Errorf("UpdateAndRemoveSchedule delete fixed_date - %w", err)
	}
	if _, err := tx.Exec(ctx, `DELETE FROM schedule WHERE task_id = $1`, task.ID); err != nil {
		return nil, fmt.Errorf("UpdateAndRemoveSchedule delete schedule - %w", err)
	}

	if err := tx.Commit(ctx); err != nil {
		return nil, fmt.Errorf("UpdateAndRemoveSchedule commit - %w", err)
	}

	return updated, nil
}

func (r *Repository) List(ctx context.Context, limit, offset int) ([]taskdomain.Task, error) {
	const query = `
		SELECT id, title, description, status, created_at, updated_at
		FROM tasks
		ORDER BY id DESC
		LIMIT $1 OFFSET $2
	`

	rows, err := r.pool.Query(ctx, query, limit, offset)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	tasks := make([]taskdomain.Task, 0)
	for rows.Next() {
		task, err := scanTask(rows)
		if err != nil {
			return nil, err
		}

		task.Schedule, err = r.GetSchedule(ctx, task.ID)
		if err != nil {
			return nil, err
		}

		tasks = append(tasks, *task)
	}

	if err := rows.Err(); err != nil {
		return nil, err
	}

	return tasks, nil
}

type taskMutationDTO struct {
	Title       string                   `json:"title"`
	Description string                   `json:"description"`
	Status      taskdomain.Status        `json:"status"`
	Schedule    *taskMutationScheduleDTO `json:"schedule"`
}

type taskMutationScheduleDTO struct {
	Type              int      `json:"type"`
	AfterAmountOfDays int      `json:"after_amount_of_days"`
	DayOfMonth        int      `json:"day_of_month"`
	OddOrEven         string   `json:"odd_or_even"`
	CertainDays       []string `json:"certain_days"`
}

func (d *taskMutationScheduleDTO) toDomain() (*taskdomain.Schedule, error) {
	schedule := &taskdomain.Schedule{
		Type:              d.Type,
		AfterAmountOfDays: &d.AfterAmountOfDays,
		DayOfMonth:        &d.DayOfMonth,
		OddOrEven:         &d.OddOrEven,
		CertainDays:       make([]time.Time, 0),
	}

	layout := "2006-01-02"
	for _, day := range d.CertainDays {
		t, err := time.Parse(layout, day)
		if err != nil {
			return nil, err
		}
		schedule.CertainDays = append(schedule.CertainDays, t)
	}

	return schedule, nil
}

type taskDTO struct {
	ID          int64              `json:"id"`
	Title       string             `json:"title"`
	Description string             `json:"description"`
	Status      taskdomain.Status  `json:"status"`
	CreatedAt   time.Time          `json:"created_at"`
	UpdatedAt   time.Time          `json:"updated_at"`
	Schedule    *taskScheduleDTO   `json:"schedule,omitempty"`
}

type taskScheduleDTO struct {
	Type              int      `json:"type"`
	AfterAmountOfDays int      `json:"after_amount_of_days,omitempty"`
	DayOfMonth        int      `json:"day_of_month,omitempty"`
	OddOrEven         string   `json:"odd_or_even,omitempty"`
	CertainDays       []string `json:"certain_days,omitempty"`
}

func newTaskScheduleDTO(s *taskdomain.Schedule) *taskScheduleDTO {
	if s == nil {
		return nil
	}

	d := &taskScheduleDTO{Type: s.Type}
	if s.AfterAmountOfDays != nil {
		d.AfterAmountOfDays = *s.AfterAmountOfDays
	}
	if s.DayOfMonth != nil {
		d.DayOfMonth = *s.DayOfMonth
	}
	if s.OddOrEven != nil {
		d.OddOrEven = *s.OddOrEven
	}

	layout := "2006-01-02"
	for _, t := range s.CertainDays {
		d.CertainDays = append(d.CertainDays, t.UTC().Format(layout))
	}

	return d
}

func newTaskDTO(task *taskdomain.Task) taskDTO {
	return taskDTO{
		ID:          task.ID,
		Title:       task.Title,
		Description: task.Description,
		Status:      task.Status,
		CreatedAt:   task.CreatedAt,
		UpdatedAt:   task.UpdatedAt,
		Schedule:    newTaskScheduleDTO(task.Schedule),
	}
}

type patchTaskDTO struct {
	Title        *string            `json:"title"`
	Description  *string            `json:"description"`
	Status       *taskdomain.Status `json:"status"`
	ScheduleJSON *json.RawMessage   `json:"schedule"`
}

func (d *patchTaskDTO) toPatchInput() (taskusecase.PatchInput, error) {
	out := taskusecase.PatchInput{
		Title:       d.Title,
		Description: d.Description,
		Status:      d.Status,
	}
	if d.ScheduleJSON == nil {
		return out, nil
	}
	raw := strings.TrimSpace(string(*d.ScheduleJSON))
	if raw == "null" {
		out.RemoveSchedule = true
		return out, nil
	}
	var inner taskMutationScheduleDTO
	if err := json.Unmarshal([]byte(raw), &inner); err != nil {
		return taskusecase.PatchInput{}, err
	}
	sch, err := inner.toDomain()
	if err != nil {
		return taskusecase.PatchInput{}, err
	}
	out.Schedule = sch
	return out, nil
}

	var req taskMutationDTO
	if err := decodeJSON(r, &req); err != nil {
		writeError(w, http.StatusBadRequest, err)
		return
	}

	createInput := taskusecase.CreateInput{
		Title:       req.Title,
		Description: req.Description,
		Status:      req.Status,
	}
	if req.Schedule != nil {
		schedule, err := req.Schedule.toDomain()

			writeJSON(w, http.StatusBadRequest, "Неверный формат даты")
			return
		}

		createInput.Schedule = schedule

		writeUsecaseError(w, err)
		return

}

		schedule, err := req.Schedule.toDomain()
		if err != nil {
			writeJSON(w, http.StatusBadRequest, "Неверный формат даты")
			return
		}
		updateInput.Schedule = schedule
	}

	updated, err := h.usecase.Update(r.Context(), id, updateInput)
	if err != nil {
		writeUsecaseError(w, err)
		return
	}

	writeJSON(w, http.StatusOK, newTaskDTO(updated))
}

func (h *TaskHandler) Patch(w http.ResponseWriter, r *http.Request) {
	id, err := getIDFromRequest(r)
	if err != nil {
		writeError(w, http.StatusBadRequest, err)
		return
	}

	var req patchTaskDTO
	if err := decodeJSON(r, &req); err != nil {
		writeError(w, http.StatusBadRequest, err)
		return
	}

	in, err := req.toPatchInput()
	if err != nil {
		writeError(w, http.StatusBadRequest, err)
		return
	}

	updated, err := h.usecase.Patch(r.Context(), id, in)
	if err != nil {
		writeUsecaseError(w, err)
		return
	}

	writeJSON(w, http.StatusOK, newTaskDTO(updated))
}

func (h *TaskHandler) NextRun(w http.ResponseWriter, r *http.Request) {
	id, err := getIDFromRequest(r)
	if err != nil {
		writeError(w, http.StatusBadRequest, err)
		return
	}

	raw := r.URL.Query().Get("date")
	if raw == "" {
		writeError(w, http.StatusBadRequest, errors.New("missing date query parameter (YYYY-MM-DD)"))
		return
	}

	ref, err := time.ParseInLocation("2006-01-02", raw, time.UTC)
	if err != nil {
		writeError(w, http.StatusBadRequest, errors.New("invalid date: use YYYY-MM-DD"))
		return
	}

	next, err := h.usecase.NextRun(r.Context(), id, ref)
	if err != nil {
		writeUsecaseError(w, err)
		return
	}

	writeJSON(w, http.StatusOK, nextRunResponseDTO{NextRun: next.UTC().Format("2006-01-02")})
}

func getIDFromRequest(r *http.Request) (int64, error) {
	rawID := mux.Vars(r)["id"]
	if rawID == "" {
		return 0, errors.New("missing task id")
	}

	id, err := strconv.ParseInt(rawID, 10, 64)
	if err != nil {
		return 0, errors.New("invalid task id")
	}

	if id <= 0 {
		return 0, errors.New("invalid task id")
	}

	return id, nil
}

func decodeJSON(r *http.Request, dst any) error {
	decoder := json.NewDecoder(r.Body)
	decoder.DisallowUnknownFields()

	if err := decoder.Decode(dst); err != nil {
		return err
	}

	return nil
}

func writeUsecaseError(w http.ResponseWriter, err error) {
	switch {
	case errors.Is(err, taskdomain.ErrNotFound):
		writeError(w, http.StatusNotFound, err)
	case errors.Is(err, taskdomain.ErrNoSchedule), errors.Is(err, taskdomain.ErrNoNextRun):
		writeError(w, http.StatusNotFound, err)
	case errors.Is(err, taskusecase.ErrInvalidInput):
		writeError(w, http.StatusBadRequest, err)
	default:
		writeError(w, http.StatusInternalServerError, err)
	}
}

func writeError(w http.ResponseWriter, status int, err error) {
	writeJSON(w, status, map[string]string{
		"error": err.Error(),
	})
}

func writeJSON(w http.ResponseWriter, status int, payload any) {
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(status)

	_ = json.NewEncoder(w).Encode(payload)
}

type Service struct {
	repo Repository

func NewService(repo Repository) *Service {
	return &Service{
		repo: repo,
		now:  func() time.Time { return time.Now().UTC() },
	}
}

func (s *Service) Create(ctx context.Context, input CreateInput) (*taskdomain.Task, error) {
	normalized, err := validateCreateInput(input)
	if err != nil {
		return nil, err
	}

	model := &taskdomain.Task{
		Title:       normalized.Title,
		Description: normalized.Description,
		Status:      normalized.Status,
	}
	now := s.now()
	model.CreatedAt = now
	model.UpdatedAt = now

	var created *taskdomain.Task
	if input.Schedule != nil {
		created, err = s.repo.CreateWithSchedule(ctx, model, input.Schedule)
	} else {
		created, err = s.repo.Create(ctx, model)
	}
	if err != nil {
		return nil, err
	}

	created.Schedule, err = s.repo.GetSchedule(ctx, created.ID)
	if err != nil {
		return nil, err
	}

	return created, nil
}

		return nil, fmt.Errorf("%w: id must be positive", ErrInvalidInput)
	}

	return s.repo.GetByID(ctx, id)
}

// NextRun returns the next schedule execution date strictly after ref (calendar day, UTC).
func (s *Service) NextRun(ctx context.Context, id int64, ref time.Time) (time.Time, error) {
	if id <= 0 {
		return time.Time{}, fmt.Errorf("%w: id must be positive", ErrInvalidInput)
	}

	t, err := s.repo.GetByID(ctx, id)
	if err != nil {
		return time.Time{}, err
	}
	if t.Schedule == nil {
		return time.Time{}, taskdomain.ErrNoSchedule
	}

	next, ok := taskdomain.NextExecutionAfter(t.CreatedAt, t.Schedule, ref)
	if !ok {
		return time.Time{}, taskdomain.ErrNoNextRun
	}

	return next, nil
}

func (s *Service) Update(ctx context.Context, id int64, input UpdateInput) (*taskdomain.Task, error) {
	if id <= 0 {
		return nil, fmt.Errorf("%w: id must be positive", ErrInvalidInput)
	}

	normalized, err := validateUpdateInput(input)
	if err != nil {
		return nil, err
	}

	model := &taskdomain.Task{
		ID:          id,
		Title:       normalized.Title,
		Description: normalized.Description,

		UpdatedAt:   s.now(),
	}

	var updated *taskdomain.Task
	if input.Schedule != nil {
		input.Schedule.TaskID = id
		updated, err = s.repo.UpdateWithReplaceSchedule(ctx, model, input.Schedule)
	} else {
		updated, err = s.repo.Update(ctx, model)
	}
	if err != nil {

	}

	updated.Schedule, err = s.repo.GetSchedule(ctx, updated.ID)
	if err != nil {
		return nil, err
	}

}

func (s *Service) Patch(ctx context.Context, id int64, input PatchInput) (*taskdomain.Task, error) {
	if id <= 0 {
		return nil, fmt.Errorf("%w: id must be positive", ErrInvalidInput)
	}

	if !patchHasChanges(input) {
		return nil, fmt.Errorf("%w: nothing to patch", ErrInvalidInput)
	}

	normalized, err := validatePatchInput(input)
	if err != nil {
		return nil, err
	}

	cur, err := s.repo.GetByID(ctx, id)
	if err != nil {
		return nil, err
	}

	merged := *cur
	if normalized.Title != nil {
		merged.Title = *normalized.Title
	}
	if normalized.Description != nil {
		merged.Description = *normalized.Description
	}
	if normalized.Status != nil {
		merged.Status = *normalized.Status
	}
	merged.UpdatedAt = s.now()

	var updated *taskdomain.Task
	switch {
	case normalized.RemoveSchedule:
		updated, err = s.repo.UpdateAndRemoveSchedule(ctx, &merged)
	case normalized.Schedule != nil:
		normalized.Schedule.TaskID = id
		updated, err = s.repo.UpdateWithReplaceSchedule(ctx, &merged, normalized.Schedule)
	default:
		updated, err = s.repo.Update(ctx, &merged)
	}
	if err != nil {
		return nil, err
	}

	updated.Schedule, err = s.repo.GetSchedule(ctx, updated.ID)
	if err != nil {
		return nil, err
	}

	return updated, nil
}

func patchHasChanges(in PatchInput) bool {
	return in.Title != nil || in.Description != nil || in.Status != nil ||
		in.Schedule != nil || in.RemoveSchedule
}

func validatePatchInput(input PatchInput) (PatchInput, error) {
	if input.RemoveSchedule && input.Schedule != nil {
		return PatchInput{}, fmt.Errorf("%w: cannot set schedule and remove schedule", ErrInvalidInput)
	}

	if input.Title != nil {
		t := strings.TrimSpace(*input.Title)
		if t == "" {
			return PatchInput{}, fmt.Errorf("%w: title cannot be empty", ErrInvalidInput)
		}
		input.Title = &t
	}
	if input.Description != nil {
		d := strings.TrimSpace(*input.Description)
		input.Description = &d
	}
	if input.Status != nil && !input.Status.Valid() {
		return PatchInput{}, fmt.Errorf("%w: invalid status", ErrInvalidInput)
	}

	return input, nil
}

	if err != nil {

}
